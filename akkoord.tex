\newcommand{\akkoord}[2] {
  \begin{tikzpicture}[font=\sffamily\sansmath]
  % Draw snares and frets (6 and 4)
  \draw[help lines] (0,0) grid (5,4);
  % Write name of chord
  \draw (2.5,5.5) node {\Huge #2};
  % Draw fingers
  \foreach \x/\y/\z [count=\i] in {#1} {
    \ifx\x\mute {
      \draw (\i-1, 4.5) node {\Large X};
    } \else {
      \ifx\x\open {
        \draw (\i-1, 4.5) node {\Large O};
        \ifx\y\empty {
          \relax
        } \else {
          \draw (\i-1, -0.5) node {\Large \y};
        } \fi
      } \else {
        \filldraw [fill=white] (\i-1, 4.5-\x) circle [radius=0.4];
        \draw (\i-1, 4.5-\x) node {\Large \y};
        \ifx\z\empty {
          \relax
        } \else {
          \draw (\i-1, -0.5) node {\Large \z};
        } \fi
      } \fi
    } \fi
  }
  \end{tikzpicture}
}
