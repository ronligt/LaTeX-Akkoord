\newcommand{\akkoord}[3][I] {
  \begin{tikzpicture}[font=\sffamily\sansmath,thick,scale=0.35, every node/.style={transform shape}]
  % Draw snares and frets (6 and 4)
  \draw[help lines] (0,0) grid (5,4);
  % Write position of frets
  \draw (-1, 3.5) node {\Huge #1};
  \ifx#1\base {

  } \else {
    \draw[thick] (0,4.1) -- (5,4.1);
  } \fi
  % Write name of chord
  \draw (2.5,5.5) node {\Huge #3};
  % Draw fingers
  \foreach \x/\y/\z [count=\i] in {#2} {
    \ifx\x\mute {
      \draw (\i-1, 4.5) node {\Large X};
    } \else {
      \ifx\x\open {
        \draw (\i-1, 4.5) node {\Large O};
        \ifx\y\empty {
          \relax
        } \else {
          \draw (\i-1, -0.5) node {\Large \y};
        } \fi
      } \else {
        \filldraw [fill=white] (\i-1, 4.5-\x) circle [radius=0.4];
        \draw (\i-1, 4.5-\x) node {\Large \y};
        \ifx\z\empty {
          \relax
        } \else {
          \draw (\i-1, -0.5) node {\Large \z};
        } \fi
      } \fi
    } \fi
  }
  \end{tikzpicture}
}

\makeatother
