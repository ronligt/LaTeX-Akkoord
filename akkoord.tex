\newrobustcmd{\akkoord}[3][I] {
  \begin{tikzpicture}[font=\sffamily\sansmath,scale=0.35, every node/.style={transform shape}, baseline=(current bounding box.north), my pic adjust=-2pt]
  % Draw snares and frets (6 and 4)
  \draw[help lines] (0,0) grid (5,4);
  % Write position of frets
  \draw (-1, 3.5) node {\Huge #1};
  \ifx#1\base {

  } \else {
    \draw[thick] (0,4.1) -- (5,4.1);
  } \fi
  % Write name of chord
  \draw (2.5,5.5) node {\Huge #3};
  % Draw fingers
  \foreach \x/\y/\z [count=\i] in {#2} {
    \ifx\x\mute {
      \draw (\i-1, 4.5) node {\Large X};
    } \else {
      \ifx\x\open {
        \draw (\i-1, 4.5) node {\Large O};
        \ifx\y\empty {
          \relax
        } \else {
          \draw (\i-1, -0.5) node {\Large \y};
        } \fi
      } \else {
        % Find out if we can draw a bar...
        \ifcsdef{@barLowSnare@\y} {
          \csnumgdef{@barHighSnare@\y}{\i}
          \csnumgdef{@barHighFret@\y}{\x}
        } {
          \csnumgdef{@barLowSnare@\y}{\i}
          \csnumgdef{@barLowFret@\y}{\x}
        }
        \filldraw [fill=white] (\i-1, 4.5-\x) circle [radius=0.4];
        \draw (\i-1, 4.5-\x) node {\Large \y};
        \ifx\z\empty {
          \relax
        } \else {
          \draw (\i-1, -0.5) node {\Large \z};
        } \fi
      } \fi
    } \fi
  }
  \foreach \x/\y/\z [count=\i] in {#2} {
    \ifcsdef{@barHighSnare@\y} {
      \draw[thick] (\csuse{@barLowSnare@\y}-1,4.9-\csuse{@barLowFret@\y}) to[out=20, in=160] (\csuse{@barHighSnare@\y}-1,4.9-\csuse{@barHighFret@\y});
    }
    \global\csundef{@barLowSnare@\y}
    \global\csundef{@barLowFret@\y}
    \global\csundef{@barHighSnare@\y}
    \global\csundef{@barHighFret@\y}
  }
  \end{tikzpicture}
}
